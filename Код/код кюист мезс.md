``` Python 
class Solution:
    def pivotInteger(self, n: int) -> int:
        total = n * (n + 1) // 2
        x = int(total ** 0.5)
        if x * x == total:
            return x
        return -1
```

```python
class Solution:
    def isPalindrome(self, x: int) -> bool:
      
        # Отрицательные числа не могут быть палиндромами
        if x < 0:
            return False
        
        original = x
        reversed_num = 0
        
        while x > 0:
            digit = x % 10
            reversed_num = reversed_num * 10 + digit
            x //= 10
        
        return original == reversed_num


```
