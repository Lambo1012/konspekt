```TypeScript
function addBinary(a: string, b: string): string {
    // Решение с использованием BigInt (поддерживает большие числа)
    return (BigInt('0b' + a) + BigInt('0b' + b)).toString(2);
}

// Альтернативное решение с ручной реализацией сложения поразрядно
function addBinaryManual(a: string, b: string): string {
    let result: string[] = [];
    let carry: number = 0;
    let i: number = a.length - 1;
    let j: number = b.length - 1;
    
    // Проходим строки справа налево
    while (i >= 0 || j >= 0 || carry > 0) {
        // Получаем текущие цифры (или 0 если строка закончилась)
        const digitA: number = i >= 0 ? parseInt(a[i]) : 0;
        const digitB: number = j >= 0 ? parseInt(b[j]) : 0;
        
        // Складываем текущие биты и перенос
        const sum: number = digitA + digitB + carry;
        
        // Определяем текущий бит результата и перенос
        result.push((sum % 2).toString());
        carry = Math.floor(sum / 2);
        
        // Двигаемся к следующим цифрам
        i--;
        j--;
    }
    
    // Переворачиваем массив, так как мы добавляли с конца
    return result.reverse().join('');
}

// Тестирование
console.log(addBinary("11", "1"));        // "100"
console.log(addBinary("1010", "1011"));   // "10101"

console.log(addBinaryManual("11", "1"));        // "100"
console.log(addBinaryManual("1010", "1011"));   // "10101"
```
