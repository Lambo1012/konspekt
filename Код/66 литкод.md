function plusOne(digits: number[]): number[] {
    // Идем с конца массива (с младших разрядов)
    for (let i = digits.length - 1; i >= 0; i--) {
        // Если текущая цифра меньше 9, просто увеличиваем на 1
        if (digits[i] < 9) {
            digits[i]++;
            return digits;
        }
        // Если цифра равна 9, устанавливаем в 0 и продолжаем перенос
        digits[i] = 0;
    }
    
    // Если мы дошли до начала массива и все еще есть перенос
    // (например, было [9,9,9] → [0,0,0], нужно добавить 1 в начало)
    return [1, ...digits];
}

// Альтернативное решение с явным переносом
function plusOne2(digits: number[]): number[] {
    let carry = 1; // Начинаем с переноса 1 (плюс один)
    
    for (let i = digits.length - 1; i >= 0; i--) {
        const sum = digits[i] + carry;
        digits[i] = sum % 10;
        carry = Math.floor(sum / 10);
        
        // Если переноса нет, можно завершить досрочно
        if (carry === 0) {
            return digits;
        }
    }
    
    // Если после обработки всех цифр остался перенос
    if (carry > 0) {
        digits.unshift(carry);
    }
    
    return digits;
}

// Решение с преобразованием в строку (менее эффективное, но понятное)
function plusOne3(digits: number[]): number[] {
    // Преобразуем массив цифр в большое число с помощью BigInt
    const numStr = digits.join('');
    const bigIntNum = BigInt(numStr) + BigInt(1);
    
    // Преобразуем обратно в массив цифр
    return Array.from(String(bigIntNum), Number);
}

// Тесты
console.log(plusOne([1, 2, 3]));    // [1, 2, 4]
console.log(plusOne([4, 3, 2, 1])); // [4, 3, 2, 2]
console.log(plusOne([9]));          // [1, 0]
console.log(plusOne([9, 9]));       // [1, 0, 0]
console.log(plusOne([0]));          // [1]
console.log(plusOne([1, 9, 9]));    // [2, 0, 0]
console.log(plusOne([1, 0, 9]));    // [1, 1, 0]

// Тест с большим числом
console.log(plusOne([9, 9, 9, 9, 9, 9, 9, 9])); // [1, 0, 0, 0, 0, 0, 0, 0, 0]
