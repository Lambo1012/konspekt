function mySqrt(x: number): number {
    // Бинарный поиск - оптимальное решение
    if (x < 2) return x;
    
    let left: number = 1;
    let right: number = Math.floor(x / 2);
    let result: number = 0;
    
    while (left <= right) {
        const mid: number = left + Math.floor((right - left) / 2);
        const square: number = mid * mid;
        
        if (square === x) {
            return mid;
        } else if (square < x) {
            result = mid;
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }
    
    return result;
}

// Альтернативное решение с помощью линейного поиска (медленнее для больших чисел)
function mySqrtLinear(x: number): number {
    if (x < 2) return x;
    
    for (let i: number = 1; i <= x; i++) {
        if (i * i === x) {
            return i;
        }
        if (i * i > x) {
            return i - 1;
        }
    }
    return 0;
}

// Еще одно альтернативное решение с использованием метода Ньютона (быстрое)
function mySqrtNewton(x: number): number {
    if (x < 2) return x;
    
    let guess: number = Math.floor(x / 2);
    
    while (guess * guess > x || (guess + 1) * (guess + 1) <= x) {
        guess = Math.floor((guess + Math.floor(x / guess)) / 2);
    }
    
    return guess;
}

// Тестирование
console.log(mySqrt(4));      // 2

console.log(mySqrt(1));      // 1
console.log(mySqrt(16));     // 4
console.log(mySqrt(2147395599));  // 46339 (проверка граничного случая)

// Сравнение производительности разных методов
console.log(mySqrtLinear(25));      // 5
console.log(mySqrtNewton(25));      // 5
