function lengthOfLastWord(s: string): number {

    // Убираем пробелы в начале и конце строки

    const trimmed = s.trim();

    // Разделяем строку по пробелам

    const words = trimmed.split(' ');

    // Получаем последнее слово

    const lastWord = words[words.length - 1];

    // Возвращаем длину последнего слова

    return lastWord.length;

}

  

// Альтернативное решение без split (более эффективное по памяти)

function lengthOfLastWord2(s: string): number {

    let length = 0;

    let i = s.length - 1;

    // Пропускаем пробелы в конце строки

    while (i >= 0 && s[i] === ' ') {

        i--;

    }

    // Считаем длину последнего слова

    while (i >= 0 && s[i] !== ' ') {

        length++;

        i--;

    }

    return length;

}

  

// Тесты

console.log(lengthOfLastWord("Hello World")); // 5

console.log(lengthOfLastWord("   fly me   to   the moon  ")); // 4

console.log(lengthOfLastWord("luffy is still joyboy")); // 6
