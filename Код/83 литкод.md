```python 
# Определение структуры узла связного списка
class ListNode:
    def __init__(self, val=0, next=None):
        self.val = val  # Значение узла
        self.next = next  # Ссылка на следующий узел

class Solution:
    def deleteDuplicates(self, head: ListNode) -> ListNode:
        """
        Удаляет дубликаты из отсортированного связного списка.
        
        Параметры:
        head (ListNode): Голова входного отсортированного связного списка
        
        Возвращает:
        ListNode: Голова списка без дубликатов
        """
        
        # Базовый случай: если список пустой или содержит только один элемент
        if not head or not head.next:
            return head
        
        # Устанавливаем текущий указатель на начало списка
        current = head
        
        # Проходим по списку, пока не достигнем конца
        while current and current.next:
            # Если текущее значение равно следующему значению (дубликат)
            if current.val == current.next.val:
                # Пропускаем следующий узел (удаляем дубликат)
                current.next = current.next.next
            else:
                # Переходим к следующему уникальному узлу
                current = current.next
        
        return head
```
