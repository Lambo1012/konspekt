function climbStairs(n: number): number {
    if (n <= 2) return n;
    
    let dp: number[] = new Array(n + 1);
    dp[1] = 1;
    dp[2] = 2;
    
    for (let i = 3; i <= n; i++) {
        dp[i] = dp[i - 1] + dp[i - 2];
    }
    
    return dp[n];
}

// Более оптимальное решение с O(1) по памяти:
function climbStairsOptimized(n: number): number {
    if (n <= 2) return n;
    
    let prev1 = 1; // dp[i-2]
    let prev2 = 2; // dp[i-1]
    let current = 0;
    
    for (let i = 3; i <= n; i++) {
        current = prev1 + prev2;
        prev1 = prev2;
        prev2 = current;
    }
    
    return current;
}

// Тесты:
console.log(climbStairs(2)); // 2
console.log(climbStairs(3)); // 3
console.log(climbStairs(4)); // 5
console.log(climbStairs(5)); // 8

console.log(climbStairsOptimized(2)); // 2
console.log(climbStairsOptimized(3)); // 3
console.log(climbStairsOptimized(4)); // 5
console.log(climbStairsOptimized(5)); // 8
